#!groovy
//  groovy Jenkinsfile
pipeline {
    agent { 
           label ''
        }
        docker {
            image 'ubuntu'
            args '-u root' // Запуск контейнера з правами root
        }
    
    
    stages {
        stage('Install Docker') {
            steps {
                sh 'apt-get update'
                sh 'apt-get install -y docker.io'
            }
        }
        
        stage('Install MC, Vim, Zip') {
            steps {
                sh 'apt-get update'
                sh 'apt-get install -y mc vim zip'
            }
        }
    }
}
